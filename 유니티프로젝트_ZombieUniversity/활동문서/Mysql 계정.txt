#DB아이디 조회 MYsQL 8.0Command Line Client서 실행
use mysql;
select host, user, password from user;

#id생성방법 로컬버전 및 모든아이피 버전 %자리에 아이피넣어도딤
create user testId@localhost identified by '1234';
create user testId@'%' identified by '1234';

#권한부여, 플러시(적용),권한확인
GRANT ALL ON *.* TO 'testId'@'%';
flush privileges;
show grants for `root`@'%';

#권한 삭제
DELETE FROM mysql.user WHERE Host='%' AND User='testId';
flush privileges;

#유저 삭제
drop user 'testId'@localhost;
